Ejercicio de la sección 3: contador

En este ejercicio se pone en práctica lo aprendido sobre $scope y controladores en AngularJS.

En el ejercicio se crea una aplicación web que mantiene información de un contador y permite incrementarlo y decrementarlo.

Código del ejercicio

<body ng-app="contadorApp">
	!-- ... -->
	<div ng-controller="ContadorController">
		<span>{{contador}}</span>
		<button class="btn btn-primary" ng-click="incrementar()">+</button>
		<button class="btn btn-default" ng-click="decrementar()">-</button>
	</div>
	<!-- ... -->
</body>
// Declaración del módulo que se enlaza con el DOM mediante la directiva ng-app
// en la etiqueta body (puede ser otra etiqueta).
var contadorApp = angular.module("contadorApp", []);

// Declaración del controlador, que se enlaza con el DOM mediante la directiva
// ng-controller, aplicado en esta ocasión en una etiqueta DIV, aunque puede
// usarse en otra superior como body, siempre que la etiqueta donde la
// aplicamos esté contenida en la que tiene la directiva ng-app.
contadorApp.controller("ContadorController", function($scope) {
		// contador es una variable del scope que puede ser enlazada con
		// ng-model o con dobles llaves.
		$scope.contador = 0;

		// incrementar y decrementar son funciones del scope, en este caso
		// se enlazan con los botones mediante la directiva ng-click.
		$scope.incrementar = function() {
			$scope.contador += 1;
		};
		$scope.decrementar = function() {
			$scope.contador -= 1;
		};
	});
